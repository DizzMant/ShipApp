1 Работающий запрос:
SELECT ship.name
FROM ship
JOIN event ON ship.id = event.idShip
JOIN ports ON event.idPorts = ports.id
WHERE ports.city = 'Одесса'
AND event.dateArrival >= '1998-01-01'
AND event.dateArrival <= '1999-12-31';

2 Работающий запрос:
SELECT event.dateArrival, event.dateDeparture
FROM ship
JOIN event ON ship.id = event.idShip
JOIN ports ON event.idPorts = ports.id
WHERE ship.name = 'Кузнецов'
AND ports.name = 'Новороссийск'
AND event.mission = 'Починка такелажа';

3 работающий запрос:
SELECT DISTINCT country
FROM ports
WHERE country NOT IN (
    SELECT DISTINCT country
    FROM event
    INNER JOIN ports ON event.idPorts = ports.id
    WHERE mission = 'Туризм'
);

4 работающий запрос:
SELECT mission, COUNT(*) AS frequency
FROM event
INNER JOIN ports ON event.idPorts = ports.id
WHERE ports.city = 'Ялта'
GROUP BY mission
ORDER BY frequency DESC
LIMIT 1;
